1.5 Создать сервис проверки разрешений
Файл: backend/src/services/permissionService.ts
typescriptimport Permission, { PermissionResource, PermissionAction, PermissionScope } from '../models/Permission';
import RolePermission from '../models/RolePermission';
import User from '../models/User';

export interface PermissionCheckContext {
  user: User;
  targetHotelId?: string;
  targetDepartmentId?: string;
  targetUserId?: string;
}

export class PermissionService {
  /**
   * Проверить имеет ли пользователь разрешение
   */
  static async hasPermission(
    context: PermissionCheckContext,
    resource: PermissionResource,
    action: PermissionAction
  ): Promise<boolean> {
    const { user, targetHotelId, targetDepartmentId, targetUserId } = context;

    // Получить все permissions для роли пользователя
    const rolePermissions = await RolePermission.findAll({
      where: { role: user.role },
      include: [{ model: Permission }],
    });

    // Найти подходящие permissions
    for (const rp of rolePermissions) {
      const permission = rp.permission;
      
      if (permission.resource !== resource || permission.action !== action) {
        continue;
      }

      // Проверить scope
      const hasScope = await this.checkScope(user, permission.scope, {
        targetHotelId,
        targetDepartmentId,
        targetUserId,
      });

      if (hasScope) {
        return true;
      }
    }

    return false;
  }

  /**
   * Проверить соответствует ли scope
   */
  private static async checkScope(
    user: User,
    scope: PermissionScope,
    target: {
      targetHotelId?: string;
      targetDepartmentId?: string;
      targetUserId?: string;
    }
  ): Promise<boolean> {
    switch (scope) {
      case PermissionScope.ALL:
        // Доступ ко всему
        return true;

      case PermissionScope.HOTEL:
        // Доступ только к своему отелю
        if (!target.targetHotelId) return true; // Если не указан target, считаем что это свой отель
        return user.hotelId === target.targetHotelId;

      case PermissionScope.DEPARTMENT:
        // Доступ только к своему департаменту
        if (!target.targetDepartmentId) return true;
        return user.departmentId === target.targetDepartmentId;

      case PermissionScope.OWN:
        // Доступ только к своим записям
        if (!target.targetUserId) return true;
        return user.id === target.targetUserId;

      default:
        return false;
    }
  }

  /**
   * Получить список доступных действий для ресурса
   */
  static async getAvailableActions(
    user: User,
    resource: PermissionResource
  ): Promise<PermissionAction[]> {
    const rolePermissions = await RolePermission.findAll({
      where: { role: user.role },
      include: [{ 
        model: Permission,
        where: { resource }
      }],
    });

    return rolePermissions.map(rp => rp.permission.action);
  }

  /**
   * Проверить может ли пользователь управлять другим пользователем
   */
  static async canManageUser(
    actor: User,
    targetUser: User
  ): Promise<boolean> {
    // Нельзя управлять самим собой через эту функцию
    if (actor.id === targetUser.id) return false;

    // SUPER_ADMIN может всех
    if (actor.role === 'SUPER_ADMIN') return true;

    // HOTEL_ADMIN может пользователей своего отеля
    if (actor.role === 'HOTEL_ADMIN' && actor.hotelId === targetUser.hotelId) {
      // Но не может управлять другими HOTEL_ADMIN или SUPER_ADMIN
      if (targetUser.role === 'HOTEL_ADMIN' || targetUser.role === 'SUPER_ADMIN') {
        return false;
      }
      return true;
    }

    return false;
  }
}
1.6 Создать middleware для проверки permissions
Файл: backend/src/middleware/permission.ts
typescriptimport { Request, Response, NextFunction } from 'express';
import { PermissionService } from '../services/permissionService';
import { PermissionResource, PermissionAction } from '../models/Permission';

/**
 * Расширение Request для добавления user
 */
declare global {
  namespace Express {
    interface Request {
      user?: any;
    }
  }
}

/**
 * Middleware для проверки permissions
 */
export const requirePermission = (
  resource: PermissionResource,
  action: PermissionAction,
  options?: {
    getTargetHotelId?: (req: Request) => string | undefined;
    getTargetDepartmentId?: (req: Request) => string | undefined;
    getTargetUserId?: (req: Request) => string | undefined;
  }
) => {
  return async (req: Request, res: Response, next: NextFunction) => {
    try {
      if (!req.user) {
        return res.status(401).json({ error: 'Authentication required' });
      }

      const context = {
        user: req.user,
        targetHotelId: options?.getTargetHotelId?.(req),
        targetDepartmentId: options?.getTargetDepartmentId?.(req),
        targetUserId: options?.getTargetUserId?.(req),
      };

      const hasPermission = await PermissionService.hasPermission(
        context,
        resource,
        action
      );

      if (!hasPermission) {
        return res.status(403).json({ 
          error: 'Forbidden',
          message: `You don't have permission to ${action} ${resource}` 
        });
      }

      next();
    } catch (error) {
      console.error('Permission check error:', error);
      res.status(500).json({ error: 'Internal server error' });
    }
  };
};

/**
 * Middleware для проверки может ли пользователь управлять другим пользователем
 */
export const requireCanManageUser = async (
  req: Request,
  res: Response,
  next: NextFunction
) => {
  try {
    if (!req.user) {
      return res.status(401).json({ error: 'Authentication required' });
    }

    const targetUserId = req.params.userId || req.body.userId;
    if (!targetUserId) {
      return res.status(400).json({ error: 'Target user ID required' });
    }

    const targetUser = await User.findByPk(targetUserId);
    if (!targetUser) {
      return res.status(404).json({ error: 'User not found' });
    }

    const canManage = await PermissionService.canManageUser(req.user, targetUser);
    if (!canManage) {
      return res.status(403).json({ 
        error: 'Forbidden',
        message: 'You cannot manage this user' 
      });
    }

    // Прикрепить targetUser к request для использования в контроллере
    req.targetUser = targetUser;
    next();
  } catch (error) {
    console.error('User management check error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
};

typescriptimport express from 'express';
import { requireAuth } from '../middleware/auth';
import { requirePermission, requireCanManageUser } from '../middleware/permission';
import { PermissionResource, PermissionAction } from '../models/Permission';
import * as userController from '../controllers/user';

const router = express.Router();

// Просмотр пользователей - требует permission read users в рамках hotel
router.get(
  '/',
  requireAuth,
  requirePermission(PermissionResource.USERS, PermissionAction.READ, {
    getTargetHotelId: (req) => req.query.hotelId as string || req.user.hotelId
  }),
  userController.listUsers
);

// Создание пользователя - требует permission create users в рамках hotel
router.post(
  '/',
  requireAuth,
  requirePermission(PermissionResource.USERS, PermissionAction.CREATE, {
    getTargetHotelId: (req) => req.body.hotelId || req.user.hotelId
  }),
  userController.createUser
);

// Обновление пользователя - требует возможность управлять пользователем
router.patch(
  '/:userId',
  requireAuth,
  requireCanManageUser,
  userController.updateUser
);

// Блокировка пользователя - требует update permission
router.post(
  '/:userId/block',
  requireAuth,
  requireCanManageUser,
  userController.blockUser
);

// Удаление пользователя - требует delete permission
router.delete(
  '/:userId',
  requireAuth,
  requireCanManageUser,
  userController.deleteUser
);

export default router;

#### 2.6 Создать базовый query helper для фильтрации по контексту

Файл: `backend/src/utils/contextQuery.ts`
```typescript
import { WhereOptions } from 'sequelize';
import User from '../models/User';

/**
 * Создать where условие с учётом контекста пользователя
 */
export function buildContextWhere(
  user: User,
  additionalWhere?: WhereOptions
): WhereOptions {
  const baseWhere: WhereOptions = {};

  // SUPER_ADMIN видит всё
  if (user.role === 'SUPER_ADMIN') {
    return { ...baseWhere, ...additionalWhere };
  }

  // HOTEL_ADMIN видит свой отель
  if (user.role === 'HOTEL_ADMIN') {
    baseWhere.hotelId = user.hotelId;
    return { ...baseWhere, ...additionalWhere };
  }

  // DEPARTMENT_MANAGER и USER видят свой департамент
  baseWhere.hotelId = user.hotelId;
  baseWhere.departmentId = user.departmentId;

  return { ...baseWhere, ...additionalWhere };
}

/**
 * Проверить может ли пользователь получить доступ к ресурсу
 */
export function canAccessResource(
  user: User,
  resource: { hotelId: string; departmentId?: string }
): boolean {
  if (user.role === 'SUPER_ADMIN') return true;
  if (user.hotelId !== resource.hotelId) return false;
  if (user.role === 'HOTEL_ADMIN') return true;
  if (resource.departmentId && user.departmentId !== resource.departmentId) {
    return false;
  }
  return true;
}
```

#### 3.1 Создать сервис для расчёта статусов срока годности

Файл: `backend/src/services/expiryService.ts`
```typescript
import { differenceInDays, isPast, parseISO } from 'date-fns';

export enum ExpiryStatus {
  OK = 'OK',
  WARNING = 'WARNING',
  CRITICAL = 'CRITICAL',
  EXPIRED = 'EXPIRED',
}

export interface ExpiryStatusColor {
  status: ExpiryStatus;
  color: string;
  daysUntilExpiry: number;
}

/**
 * Настройки порогов по умолчанию
 * В будущем будут загружаться из settings
 */
const DEFAULT_THRESHOLDS = {
  warning: 7,   // За 7 дней до истечения
  critical: 3,  // За 3 дня до истечения
};

export class ExpiryService {
  /**
   * Рассчитать статус срока годности
   */
  static calculateStatus(
    expiryDate: Date | string,
    thresholds = DEFAULT_THRESHOLDS
  ): ExpiryStatusColor {
    const expiry = typeof expiryDate === 'string' ? parseISO(expiryDate) : expiryDate;
    const now = new Date();
    const daysUntilExpiry = differenceInDays(expiry, now);

    // Уже истёк
    if (isPast(expiry)) {
      return {
        status: ExpiryStatus.EXPIRED,
        color: 'gray',
        daysUntilExpiry,
      };
    }

    // Критично (скоро истечёт)
    if (daysUntilExpiry <= thresholds.critical) {
      return {
        status: ExpiryStatus.CRITICAL,
        color: 'red',
        daysUntilExpiry,
      };
    }

    // Внимание
    if (daysUntilExpiry <= thresholds.warning) {
      return {
        status: ExpiryStatus.WARNING,
        color: 'yellow',
        daysUntilExpiry,
      };
    }

    // В норме
    return {
      status: ExpiryStatus.OK,
      color: 'green',
      daysUntilExpiry,
    };
  }

  /**
   * Получить текстовое описание статуса
   */
  static getStatusText(status: ExpiryStatus, daysUntilExpiry: number): string {
    switch (status) {
      case ExpiryStatus.EXPIRED:
        return `Expired ${Math.abs(daysUntilExpiry)} days ago`;
      case ExpiryStatus.CRITICAL:
        return `Expires in ${daysUntilExpiry} days`;
      case ExpiryStatus.WARNING:
        return `Expires in ${daysUntilExpiry} days`;
      case ExpiryStatus.OK:
        return `${daysUntilExpiry} days remaining`;
      default:
        return 'Unknown status';
    }
  }

  /**
   * Получить порог для статуса
   */
  static getThresholdForStatus(status: ExpiryStatus): number {
    switch (status) {
      case ExpiryStatus.WARNING:
        return DEFAULT_THRESHOLDS.warning;
      case ExpiryStatus.CRITICAL:
        return DEFAULT_THRESHOLDS.critical;
      default:
        return 0;
    }
  }
}
```

```typescript
/**
 * Собрать партию (уменьшить quantity)
 */
export async function collectBatch(req: Request, res: Response) {
  try {
    const user = req.user;
    const { batchId } = req.params;
    const { quantity } = req.body;

    // Найти партию с продуктом и категорией
    const batch = await Batch.findByPk(batchId, {
      include: [
        {
          model: Product,
          as: 'product',
          include: [
            {
              model: Category,
              as: 'category',
            },
          ],
        },
      ],
    });

    if (!batch) {
      return res.status(404).json({ error: 'Batch not found' });
    }

    // Проверка прав доступа
    if (!canAccessResource(user, batch)) {
      return res.status(403).json({ error: 'Access denied' });
    }

    // Проверка quantity
    if (quantity > batch.quantity) {
      return res.status(400).json({ error: 'Quantity exceeds available stock' });
    }

    const snapshotBefore = batch.toJSON();

    // Создать запись в истории сборов СО SNAPSHOT данных
    await CollectionHistory.create({
      batchId: batch.id,
      productId: batch.productId,
      hotelId: batch.hotelId,
      departmentId: batch.departmentId,
      userId: user.id,
      quantityCollected: quantity,
      quantityRemaining: batch.quantity - quantity,
      expiryDate: batch.expiryDate,           // SNAPSHOT
      productName: batch.product.name,        // SNAPSHOT
      categoryName: batch.product.category.name, // SNAPSHOT
    });

    // Обновить партию
    batch.quantity -= quantity;

    // Если quantity = 0, можно удалить партию
    if (batch.quantity === 0) {
      await batch.destroy();
    } else {
      await batch.save();
    }

    // Логировать в audit log
    await AuditService.log({
      user,
      action: ActivityAction.COLLECT,
      entityType: ActivityEntityType.BATCH,
      entityId: batch.id,
      entityName: batch.product.name,
      snapshotBefore,
      snapshotAfter: batch.quantity === 0 ? null : batch.toJSON(),
      request: req,
    });

    res.json({ 
      message: 'Batch collected successfully',
      batch: batch.quantity > 0 ? batch : null,
    });
  } catch (error) {
    console.error('Error collecting batch:', error);
    res.status(500).json({ error: 'Failed to collect batch' });
  }
}
```

#### 5.3 Создать NotificationService

Файл: `backend/src/services/notificationService.ts`
```typescript
import Notification, { NotificationPriority, NotificationStatus } from '../models/Notification';
import NotificationRule, { NotificationRuleType, NotificationChannel } from '../models/NotificationRule';
import User from '../models/User';
import Batch from '../models/Batch';
import { ExpiryService, ExpiryStatus } from './expiryService';
import { TelegramService } from './telegramService';
import { addHours } from 'date-fns';

export class NotificationService {
  /**
   * Проверить все партии и создать уведомления
   */
  static async checkExpiringBatches(): Promise<void> {
    console.log('Checking expiring batches...');

    // Получить все активные правила expiry
    const rules = await NotificationRule.findAll({
      where: {
        type: NotificationRuleType.EXPIRY,
        enabled: true,
      },
    });

    for (const rule of rules) {
      await this.processExpiryRule(rule);
    }
  }

  /**
   * Обработать правило expiry
   */
  private static async processExpiryRule(rule: NotificationRule): Promise<void> {
    const { hotelId, departmentId, warningDays = 7, criticalDays = 3 } = rule;

    // Получить партии в контексте правила
    const where: any = {};
    if (hotelId) where.hotelId = hotelId;
    if (departmentId) where.departmentId = departmentId;

    const batches = await Batch.findAll({
      where,
      include: [{ model: Product, as: 'product' }],
    });

    for (const batch of batches) {
      const { status, daysUntilExpiry } = ExpiryService.calculateStatus(batch.expiryDate, {
        warning: warningDays,
        critical: criticalDays,
      });

      // Пропустить OK и EXPIRED
      if (status === ExpiryStatus.OK || status === ExpiryStatus.EXPIRED) {
        continue;
      }

      // Проверить не отправляли ли уже уведомление недавно
      const recentNotification = await Notification.findOne({
        where: {
          type: NotificationRuleType.EXPIRY,
          'data.batchId': batch.id,
          createdAt: {
            [Op.gte]: new Date(Date.now() - 24 * 60 * 60 * 1000), // За последние 24 часа
          },
        },
      });

      if (recentNotification) {
        continue; // Уже отправляли
      }

      // Определить priority
      const priority = status === ExpiryStatus.CRITICAL 
        ? NotificationPriority.CRITICAL 
        : NotificationPriority.MEDIUM;

      // Создать уведомления для получателей
      await this.createNotificationsForRecipients(rule, {
        type: NotificationRuleType.EXPIRY,
        priority,
        title: `Product expiring soon: ${batch.product.name}`,
        message: `${batch.product.name} will expire in ${daysUntilExpiry} days (${batch.expiryDate.toLocaleDateString()})`,
        data: {
          batchId: batch.id,
          productId: batch.productId,
          expiryDate: batch.expiryDate,
          daysUntilExpiry,
          status,
        },
        hotelId: batch.hotelId,
        departmentId: batch.departmentId,
      });
    }
  }

  /**
   * Создать уведомления для всех получателей по правилу
   */
  private static async createNotificationsForRecipients(
    rule: NotificationRule,
    notificationData: {
      type: NotificationRuleType;
      priority: NotificationPriority;
      title: string;
      message: string;
      data: any;
      hotelId: string;
      departmentId?: string;
    }
  ): Promise<void> {
    // Найти получателей
    const recipients = await this.findRecipients(rule);

    for (const user of recipients) {
      for (const channel of rule.channels) {
        await Notification.create({
          ...notificationData,
          recipientUserId: user.id,
          channel,
          status: NotificationStatus.PENDING,
          retryCount: 0,
        });
      }
    }
  }

  /**
   * Найти получателей по правилу
   */
  private static async findRecipients(rule: NotificationRule): Promise<User[]> {
    const where: any = {};

    if (rule.hotelId) where.hotelId = rule.hotelId;
    if (rule.departmentId) where.departmentId = rule.departmentId;

    // Фильтр по ролям
    if (rule.recipientRoles.length > 0) {
      where.role = { [Op.in]: rule.recipientRoles };
    }

    // Конкретные пользователи
    if (rule.recipientUserIds && rule.recipientUserIds.length > 0) {
      where[Op.or] = [
        { role: { [Op.in]: rule.recipientRoles } },
        { id: { [Op.in]: rule.recipientUserIds } },
      ];
    }

    return await User.findAll({ where });
  }

  /**
   * Отправить pending уведомления
   */
  static async sendPendingNotifications(): Promise<void> {
    console.log('Sending pending notifications...');

    const notifications = await Notification.findAll({
      where: {
        status: NotificationStatus.PENDING,
        [Op.or]: [
          { nextRetryAt: null },
          { nextRetryAt: { [Op.lte]: new Date() } },
        ],
      },
      include: [{ model: User, as: 'recipient' }],
      limit: 100, // Обрабатывать порциями
    });

    for (const notification of notifications) {
      await this.sendNotification(notification);
    }
  }

  /**
   * Отправить конкретное уведомление
   */
  private static async sendNotification(notification: Notification): Promise<void> {
    try {
      switch (notification.channel) {
        case NotificationChannel.APP:
          // APP уведомления просто меняют статус на SENT
          // Frontend будет их читать через API
          await notification.update({
            status: NotificationStatus.SENT,
            sentAt: new Date(),
          });
          break;

        case NotificationChannel.TELEGRAM:
          await this.sendTelegramNotification(notification);
          break;

        case NotificationChannel.EMAIL:
          await this.sendEmailNotification(notification);
          break;
      }

      console.log(`Notification ${notification.id} sent successfully`);
    } catch (error) {
      console.error(`Failed to send notification ${notification.id}:`, error);

      // Retry logic
      const maxRetries = 3;
      if (notification.retryCount < maxRetries) {
        await notification.update({
          status: NotificationStatus.PENDING,
          retryCount: notification.retryCount + 1,
          nextRetryAt: addHours(new Date(), Math.pow(2, notification.retryCount)), // Exponential backoff
          failureReason: error.message,
        });
      } else {
        await notification.update({
          status: NotificationStatus.FAILED,
          failureReason: error.message,
        });
      }
    }
  }

  /**
   * Отправить Telegram уведомление
   */
  private static async sendTelegramNotification(notification: Notification): Promise<void> {
    const user = notification.recipient;
    
    if (!user.telegramChatId) {
      throw new Error('User does not have Telegram chat ID');
    }

    await TelegramService.sendMessage(user.telegramChatId, {
      title: notification.title,
      message: notification.message,
      priority: notification.priority,
    });

    await notification.update({
      status: NotificationStatus.SENT,
      sentAt: new Date(),
    });
  }

   * Отправить Email уведомление
   */
  private static async sendEmailNotification(notification: Notification): Promise<void> {

  /**
   * Получить уведомления пользователя
   */
  static async getUserNotifications(
    userId: string,
    options?: {
      unreadOnly?: boolean;
      limit?: number;
      offset?: number;
    }
  ): Promise<{ notifications: Notification[]; total: number }> {
    const where: any = {
      recipientUserId: userId,
      channel: NotificationChannel.APP, // Только APP уведомления
    };

    if (options?.unreadOnly) {
      where.readAt = null;
    }

    const { count, rows } = await Notification.findAndCountAll({
      where,
      order: [['createdAt', 'DESC']],
      limit: options?.limit || 50,
      offset: options?.offset || 0,
    });

    return { notifications: rows, total: count };
  }

  /**
   * Отметить уведомление как прочитанное
   */
  static async markAsRead(notificationId: string, userId: string): Promise<void> {
    const notification = await Notification.findOne({
      where: {
        id: notificationId,
        recipientUserId: userId,
      },
    });

    if (!notification) {
      throw new Error('Notification not found');
    }

      status: NotificationStatus.READ,
```

Файл: `backend/src/services/filterService.ts`
```typescript
import { Op, WhereOptions } from 'sequelize';
import { parseISO } from 'date-fns';

export interface CommonFilters {
  hotelId?: string;
  departmentIds?: string[];
  dateFrom?: Date | string;
  dateTo?: Date;
  categoryIds?: string[];
  status?: string[];
  search?: string;
  limit?: number;
  offset?: number;
}

export interface PaginationMeta {
  total: number;
  limit: number;
  offset: number;
  hasNext: boolean;
  hasPrev: boolean;
  page: number;
  totalPages: number;
}

export class FilterService {
  /**
   * Парсинг фильтров из query params
   */
  static parseFilters(query: any): CommonFilters {
    const filters: CommonFilters = {};

    if (query.hotelId) filters.hotelId = query.hotelId;
    
    if (query.departmentIds) {
      filters.departmentIds = Array.isArray(query.departmentIds) 
        ? query.departmentIds 
        : [query.departmentIds];
    }

    if (query.dateFrom) {
      filters.dateFrom = typeof query.dateFrom === 'string' 
        ? parseISO(query.dateFrom) 
        : query.dateFrom;
    }

    if (query.dateTo) {
      filters.dateTo = typeof query.dateTo === 'string' 
        ? parseISO(query.dateTo) 
        : query.dateTo;
    }

    if (query.categoryIds) {
      filters.categoryIds = Array.isArray(query.categoryIds) 
        ? query.categoryIds 
        : [query.categoryIds];
    }

    if (query.status) {
      filters.status = Array.isArray(query.status) 
        ? query.status 
        : [query.status];
    }

    if (query.search) filters.search = query.search;

    filters.limit = query.limit ? parseInt(query.limit) : 50;
    filters.offset = query.offset ? parseInt(query.offset) : 0;

    return filters;
  }

  /**
   * Построить where условие из фильтров
   */
  static buildWhere(filters: CommonFilters, baseWhere: WhereOptions = {}): WhereOptions {
    const where: WhereOptions = { ...baseWhere };

    if (filters.hotelId) {
      where.hotelId = filters.hotelId;
    }

    if (filters.departmentIds && filters.departmentIds.length > 0) {
      where.departmentId = { [Op.in]: filters.departmentIds };
    }

    if (filters.categoryIds && filters.categoryIds.length > 0) {
      // Для фильтрации по категориям нужен join с Product
      // Обрабатывается отдельно в каждом контроллере
    }

    if (filters.dateFrom || filters.dateTo) {
      where.expiryDate = {};
      if (filters.dateFrom) where.expiryDate[Op.gte] = filters.dateFrom;
      if (filters.dateTo) where.expiryDate[Op.lte] = filters.dateTo;
    }

    if (filters.search) {
      // Поиск обрабатывается отдельно через include
    }

    return where;
  }

  /**
   * Создать объект pagination
   */
  static buildPagination(total: number, limit: number, offset: number): PaginationMeta {
    const page = Math.floor(offset / limit) + 1;
    const totalPages = Math.ceil(total / limit);

    return {
      total,
      limit,
      offset,
      hasNext: offset + limit < total,
      hasPrev: offset > 0,
      page,
      totalPages,
    };
```

```typescript
import { Parser } from 'json2csv';
import ExcelJS from 'exceljs';

export enum ExportFormat {
  CSV = 'csv',
  EXCEL = 'excel',
  JSON = 'json',
}

export class ExportService {
  /**
   * Экспорт в CSV
   */
  static toCSV(data: any[], fields?: string[]): string {
    const parser = new Parser({ fields });
    return parser.parse(data);
  }

  /**
   * Экспорт в Excel
   */
  static async toExcel(data: any[], sheetName: string = 'Sheet1'): Promise<Buffer> {
    const workbook = new ExcelJS.Workbook();
    const worksheet = workbook.addWorksheet(sheetName);

    if (data.length === 0) {
      return await workbook.xlsx.writeBuffer();
    }

    // Заголовки из первого объекта
    const headers = Object.keys(data[0]);
    worksheet.columns = headers.map(key => ({
      header: key,
      key,
      width: 20,
    }));

    // Данные
    data.forEach(row => worksheet.addRow(row));

    // Стилизация заголовков
    worksheet.getRow(1).font = { bold: true };
    worksheet.getRow(1).fill = {
      type: 'pattern',
      pattern: 'solid',
      fgColor: { argb: 'FFE0E0E0' },
    };

    return await workbook.xlsx.writeBuffer();
  }

  /**
   * Экспорт в JSON
   */
  static toJSON(data: any[]): string {
    return JSON.stringify(data, null, 2);
  }
```

```typescript
import { DataTypes, Model } from 'sequelize';
import sequelize from '../config/database';

export enum SettingsScope {
  SYSTEM = 'system',
  HOTEL = 'hotel',
  DEPARTMENT = 'department',
  USER = 'user',
}

export interface SettingsAttributes {
  id: string;
  scope: SettingsScope;
  scopeId?: string; // hotelId, departmentId, или userId
  key: string;
  value: any;
  description?: string;
}

class Settings extends Model<SettingsAttributes> implements SettingsAttributes {
  public id!: string;
  public scope!: SettingsScope;
  public scopeId?: string;
  public key!: string;
  public value!: any;
  public description?: string;
}

Settings.init(
  {
    id: {
      type: DataTypes.UUID,
      defaultValue: DataTypes.UUIDV4,
      primaryKey: true,
    },
    scope: {
      type: DataTypes.ENUM(...Object.values(SettingsScope)),
      allowNull: false,
    },
    scopeId: {
      type: DataTypes.UUID,
      allowNull: true,
    },
    key: {
      type: DataTypes.STRING,
      allowNull: false,
    },
    value: {
      type: DataTypes.JSONB,
      allowNull: false,
    },
    description: {
      type: DataTypes.TEXT,
      allowNull: true,
    },
  },
  {
    sequelize,
    tableName: 'settings',
    indexes: [
      {
        unique: true,
        fields: ['scope', 'scopeId', 'key'],
      },
      { fields: ['key'] },
    ],
  }
);

export default Settings;
```

#### 7.2 Создать сервис настроек

Файл: `backend/src/services/settingsService.ts`
```typescript
import Settings, { SettingsScope } from '../models/Settings';
import User from '../models/User';

export class SettingsService {
  /**
   * Получить настройку с учётом иерархии
   * Приоритет: user > department > hotel > system
   */
  static async get(key: string, context: {
    userId?: string;
    departmentId?: string;
    hotelId?: string;
  }): Promise<any> {
    // Попробовать user scope
    if (context.userId) {
      const userSetting = await Settings.findOne({
        where: { scope: SettingsScope.USER, scopeId: context.userId, key },
      });
      if (userSetting) return userSetting.value;
    }

    // Попробовать department scope
    if (context.departmentId) {
      const deptSetting = await Settings.findOne({
        where: { scope: SettingsScope.DEPARTMENT, scopeId: context.departmentId, key },
      });
      if (deptSetting) return deptSetting.value;
    }

    // Попробовать hotel scope
    if (context.hotelId) {
      const hotelSetting = await Settings.findOne({
        where: { scope: SettingsScope.HOTEL, scopeId: context.hotelId, key },
      });
      if (hotelSetting) return hotelSetting.value;
    }

    // Fallback на system scope
    const systemSetting = await Settings.findOne({
      where: { scope: SettingsScope.SYSTEM, scopeId: null, key },
    });

    return systemSetting?.value || null;
  }

  /**
   * Установить настройку
   */
  static async set(
    key: string,
    value: any,
    scope: SettingsScope,
    scopeId?: string
  ): Promise<Settings> {
    const [setting] = await Settings.upsert({
      scope,
      scopeId,
      key,
      value,
    });

    return setting;
  }

  /**
   * Получить все настройки для scope
   */
  static async getAll(scope: SettingsScope, scopeId?: string): Promise<Record<string, any>> {
    const settings = await Settings.findAll({
      where: { scope, scopeId },
    });

    return settings.reduce((acc, s) => {
      acc[s.key] = s.value;
      return acc;
    }, {} as Record<string, any>);
  }

  /**
   * Получить настройки для пользователя (с учётом иерархии)
   */
  static async getUserSettings(user: User): Promise<Record<string, any>> {
    const keys = [
      'branding.logo',
      'branding.primaryColor',
      'branding.secondaryColor',
      'branding.companyName',
      'locale.language',
      'locale.timezone',
      'locale.dateFormat',
      'notifications.email',
      'notifications.telegram',
      'notifications.warningDays',
      'notifications.criticalDays',
    ];

    const settings: Record<string, any> = {};

    for (const key of keys) {
      settings[key] = await this.get(key, {
        userId: user.id,
        departmentId: user.departmentId,
        hotelId: user.hotelId,
      });
    }

    return settings;
  }

  /**
   * Удалить настройку
   */
  static async delete(key: string, scope: SettingsScope, scopeId?: string): Promise<void> {
    await Settings.destroy({
      where: { scope, scopeId, key },
    });
  }
}
```

#### 7.3 Создать контроллер настроек

Файл: `backend/src/controllers/settings.ts`
```typescript
import { Request, Response } from 'express';
import { SettingsService } from '../services/settingsService';
import { SettingsScope } from '../models/Settings';
import { AuditService } from '../services/auditService';
import { ActivityAction, ActivityEntityType } from '../models/ActivityLog';

/**
 * Получить настройки пользователя
 */
export async function getUserSettings(req: Request, res: Response) {
  try {
    const user = req.user;
    const settings = await SettingsService.getUserSettings(user);

    res.json({ settings });
  } catch (error) {
    console.error('Error fetching user settings:', error);
    res.status(500).json({ error: 'Failed to fetch settings' });
  }
}

/**
 * Обновить настройку
 */
export async function updateSetting(req: Request, res: Response) {
  try {
    const user = req.user;
    const { key, value, scope } = req.body;

    // Определить scopeId
    let scopeId;
    if (scope === SettingsScope.USER) {
      scopeId = user.id;
    } else if (scope === SettingsScope.DEPARTMENT) {
      scopeId = user.departmentId;
    } else if (scope === SettingsScope.HOTEL) {
      scopeId = user.hotelId;
    }

    const setting = await SettingsService.set(key, value, scope, scopeId);

    // Логировать
    await AuditService.log({
      user,
      action: ActivityAction.UPDATE,
      entityType: ActivityEntityType.SETTINGS,
      entityId: setting.id,
      entityName: key,
      snapshotAfter: { key, value, scope, scopeId },
      request: req,
    });

    res.json({ setting });
  } catch (error) {
    console.error('Error updating setting:', error);
    res.status(500).json({ error: 'Failed to update setting' });
  }
}

/**
 * Получить все настройки hotel
 */
export async function getHotelSettings(req: Request, res: Response) {
  try {
    const user = req.user;
    const settings = await SettingsService.getAll(SettingsScope.HOTEL, user.hotelId);

    res.json({ settings });
  } catch (error) {
    console.error('Error fetching hotel settings:', error);
    res.status(500).json({ error: 'Failed to fetch hotel settings' });
  }
}

/**
 * Batch update настроек
 */
export async function batchUpdateSettings(req: Request, res: Response) {
  try {
    const user = req.user;
    const { settings, scope } = req.body;

    // Определить scopeId
    let scopeId;
    if (scope === SettingsScope.USER) {
      scopeId = user.id;
    } else if (scope === SettingsScope.DEPARTMENT) {
      scopeId = user.departmentId;
    } else if (scope === SettingsScope.HOTEL) {
      scopeId = user.hotelId;
    }

    // Обновить каждую настройку
    const updatedSettings = [];
    for (const [key, value] of Object.entries(settings)) {
      const setting = await SettingsService.set(key, value, scope, scopeId);
      updatedSettings.push(setting);
    }

    // Логировать
    await AuditService.log({
      user,
      action: ActivityAction.UPDATE,
      entityType: ActivityEntityType.SETTINGS,
      entityId: 'batch',
      entityName: 'Batch Settings Update',
      snapshotAfter: { settings, scope, scopeId },
      request: req,
    });

    res.json({ settings: updatedSettings });
  } catch (error) {
    console.error('Error batch updating settings:', error);
    res.status(500).json({ error: 'Failed to batch update settings' });
  }
}
```

```typescript
import express from 'express';
import { requireAuth } from '../middleware/auth';
import { requirePermission } from '../middleware/permission';
import { PermissionResource, PermissionAction } from '../models/Permission';
import * as settingsController from '../controllers/settings';

const router = express.Router();

router.get(
  '/user',
  requireAuth,
  settingsController.getUserSettings
);

router.get(
  '/hotel',
  requireAuth,
  requirePermission(PermissionResource.SETTINGS, PermissionAction.READ),
  settingsController.getHotelSettings
);

router.put(
  '/',
  requireAuth,
  requirePermission(PermissionResource.SETTINGS, PermissionAction.UPDATE),
  settingsController.updateSetting
);

router.post(
  '/batch',
  requireAuth,
  requirePermission(PermissionResource.SETTINGS, PermissionAction.UPDATE),
  settingsController.batchUpdateSettings
);

export default router;
```
